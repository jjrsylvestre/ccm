\newcommand*{\xmin}{-2}
\newcommand*{\xmax}{2}
\newcommand*{\ymin}{-0.5}
\newcommand*{\ymax}{4}
\newcommand*{\func}{\x * \x}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt},
	xscale=1.33,
	yscale=0.75
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below left] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[right] {$f(t) = t^2$};

	\draw[green!33!black] (\xmax,2) -- (\xmin,2);

	% graph
	\draw[red, thick, domain=\xmin:0, smooth] plot (\x,{\func});
	\draw[blue, thick, domain=0:\xmax, smooth] plot (\x,{\func});

	\node[closedpt,blue] at (1.414,2) (pt) {};
	\node[closedpt,red] at (-1.414,2) (pt) {};

\end{tikzpicture}

\newcommand*{\xmin}{-0.5}
\newcommand*{\xmax}{4}
\newcommand*{\ymin}{-0.5}
\newcommand*{\ymax}{4}
\newcommand*{\func}{\x * \x}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt}
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below left] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[right] {$y$};

	% graphs
	\draw[blue, dashed, thick, domain=\xmin:0, smooth] plot (\x,{\func});
	\draw[blue, thick, domain=0:2, smooth] plot (\x,{\func}) node[left] {$t^2$};
	\draw[green!33!black, thick, domain=0:2, smooth] plot ({\func},\x) node[below] {$\sqrt{t}$};

	\draw[dashed,gray,thin] (\xmin,\xmin) -- node[above,sloped,very near end] {$y = t$} (\xmax,\xmax);
	\draw[thick,<->,red] (2,2.5) -- (2.5,2);

\end{tikzpicture}

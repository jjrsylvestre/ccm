\newcommand*{\xmin}{-0.5}
\newcommand*{\xmax}{4.5}
\newcommand*{\ymin}{-0.25}
\newcommand*{\ymax}{1}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt},
	yscale=2
]

	% rectangle fills
	\fill[blue!40!white] (0,0.5) rectangle (1,0);
	\fill[blue!40!white] (1,0.75) rectangle (3,0);
	\fill[green!33!gray] (3,0.25) rectangle (4,0);

	% grid
	\foreach \x in {-0.5,0,0.5,...,4.5} {
		\draw[very thin,gray] (\x,\ymin) -- (\x,\ymax);
	}
	\foreach \y in {-0.25,0,0.25,...,1} {
		\draw[very thin,gray] (\xmin,\y) -- (\xmax,\y);
	}

	% label grid
	\foreach \x in {1,2,3,4} {
		\node[below] at (\x,0) {$\x$};
	}
	\foreach \y in {0.5,1} {
		\node[left] at (0,\y) {$\y$};
	}

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below right] {min};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[above right] {$\mathrm{m}^3/\mathrm{min}$};

	% off
	\node[blue,thick,openpt] at (0,0) (stop) {};
	\draw[blue,thick] (\xmin,0) -- (stop);

	% medium
	\node[blue,closedpt] at (0,0.5) (mstart) {};
	\node[blue,thick,openpt] at (1,0.5) (mstop) {};
	\draw[blue,thick] (mstart) -- (mstop);

	% high
	\node[blue,closedpt] at (1,0.75) (hstart) {};
	\node[blue,thick,openpt] at (3,0.75) (hstop) {};
	\draw[blue,thick] (hstart) -- (hstop);

	% area divider
	\draw[blue,dashed]
	%   (hstart) -- (mstop)
	  (mstop) -- (1,0)
	;


	% low
	\node[green!33!black,closedpt] at (3,0.25) (start) {};
	\node[green!33!black,thick,openpt] at (4,0.25) (stop) {};
	\draw[green!33!black,thick] (start) -- (stop);

	% off
	\node[blue,closedpt] at (4,0) (start) {};
	\draw[blue,thick] (start) -- (\xmax,0);

\end{tikzpicture}

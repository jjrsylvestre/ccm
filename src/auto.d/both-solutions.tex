\newcommand*{\xmin}{0}
\newcommand*{\xmax}{10}
\newcommand*{\ymin}{0}
\newcommand*{\ymax}{10}

%
% DE is y' = r (M - y) (y - T) y
%
% We are using r = 1, T = 4, M = 8
%
% Solutions curves are
%   y = 4 [ 1 ± sqrt( A / (A - exp(-32t))) ]
% where
%   A = (y0 - 4)² / [ y0 * (y0 - 8) ]
% BUT
% have h.scaled solutions by factor of 60

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt}
]

	\draw[dashed,blue] (\xmax,1.691) -- (\xmin,1.691);
	\draw[dashed,blue] (\xmax,6.309) -- (\xmin,6.309);

	\node[left,xshift=-0.05cm] at (0,1.691) {$q_1$};
	\node[blue,xshift=-1.5cm,yshift=0.05cm] at (0,1.691) {inflection};

	\node[left,xshift=-0.05cm] at (0,6.309) {$q_2$};
	\node[blue,xshift=-1.5cm,yshift=0.05cm] at (0,6.309) {inflection};

	\draw[thick, decorate, decoration = {brace}, xshift=-0.05cm] (0,8.1) -- (0,9.85);
	\node[align=center,xshift=-1.25cm] at (0,9) {decreasing \\ concave up};

	\draw[thick, decorate, decoration = {brace}, xshift=-0.05cm] (0,6.409) -- (0,7.9);
	\node[align=center,xshift=-1.25cm] at (0,7.155) {increasing \\ concave down};

	\draw[thick, decorate, decoration = {brace}, xshift=-0.05cm] (0,4.1) -- (0,6.209);
	\node[align=center,xshift=-1.25cm] at (0,5.155) {increasing \\ concave up};

	\draw[thick, decorate, decoration = {brace}, xshift=-0.05cm] (0,1.791) -- (0,3.9);
	\node[align=center,xshift=-1.25cm] at (0,2.846) {decreasing \\ concave down};

	\draw[thick, decorate, decoration = {brace}, xshift=-0.05cm] (0,0.1) -- (0,1.591);
	\node[align=center,xshift=-1.25cm] at (0,0.845) {decreasing \\ concave up};

	% solutions with neg sqrt
	\foreach \q in {0.5,1,1.5,2,2.5,3,3.333,3.667,3.9} {
		\pgfmathparse{ (\q - 4) * (\q - 4) / (\q * \q - 8 * \q) }\edef\Acoeff{\pgfmathresult}%
		\draw[thick,blue,smooth,domain=0:9.95] plot (\x, { 4 * ( 1 - sqrt( \Acoeff / ( \Acoeff - exp(-32 * \x / 60) ) ) ) });
	}

	% solutions with pos sqrt
	\foreach \q in {4.1,4.333,4.667,5,5.5,6,6.5,7,7.5,8.5,9.15,9.8} {
		\pgfmathparse{ (\q - 4) * (\q - 4) / (\q * \q - 8 * \q) }\edef\Acoeff{\pgfmathresult}%
		\draw[thick,blue,smooth,domain=0:9.95] plot (\x, { 4 * ( 1 + sqrt( \Acoeff / ( \Acoeff - exp(-32 * \x / 60) ) ) ) });
	}

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[right] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[above] {$q$};

	\foreach \q in {0,4,8} {
		\draw[red,thick] (9.95,\q) -- (\xmin,\q);
		\node[red,xshift=-1.5cm] at (0,\q) {equilibrium};
	}

	\node[left,xshift=-0.05cm] at (0,0) {$0$};
	\node[left,xshift=-0.05cm] at (0,4) {$T$};
	\node[left,xshift=-0.05cm] at (0,8) {$M$};

\end{tikzpicture}

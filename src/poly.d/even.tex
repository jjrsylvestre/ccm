\newcommand*{\xmin}{-3}
\newcommand*{\xmax}{3}
\newcommand*{\ymin}{-0.5}
\newcommand*{\ymax}{9}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt},
	yscale=0.5
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[above right] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax);

	\draw[blue,thick,domain=-3:3,smooth] plot (\x,{\x * \x}) node[right] {$t^2$};
	\draw[green!33!black,thick,domain=-1.732:1.732,smooth] plot (\x,{\x * \x * \x * \x}) node[right] {$t^4$};
	\draw[red,thick,domain=-1.442:1.442,smooth] plot (\x,{\x * \x * \x * \x * \x * \x}) node[left] {$t^6$};

\end{tikzpicture}

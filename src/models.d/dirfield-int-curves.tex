\newcommand*{\xmin}{-0.25}
\newcommand*{\xmax}{4.25}
\newcommand*{\ymin}{-4.25}
\newcommand*{\ymax}{4.25}

%
% to maintain all segments at same length L
% use
%   delta-x = (L/2) / sqrt(1 + m^2)
%   delta-y = m * (L/2) / sqrt(1 + m^2)
% (the L/2 is because we draw two segments, each from centre out to either right or left)
%

% the rate function
\newcommand*{\dfslope}{\y * (1 - \y) / (\x + 1)}
%
\newcommand*{\halfsegmentlen}{0.15}  % (L = 0.3)
\newcommand*{\dfdeltax}{\halfsegmentlen/sqrt(1+(\dfslope)^2)}
\newcommand*{\dfdeltay}{\dfslope * \halfsegmentlen/sqrt(1+(\dfslope)^2)}

\begin{tikzpicture}[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=0.5pt}
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[above right] {$q(t)$};

	\foreach \x in {0,0.5,1,...,4} {
		\foreach \y in {-4,-3.5,-3,...,4} {
			\draw (\x,\y) -- ++($({\dfdeltax},{\dfdeltay})$);
			\draw (\x,\y) -- ++($({-\dfdeltax},{-\dfdeltay})$);
		}
	}

	% "Solutions" are
	% y / (1-y) = A * (x + 1)

	% in the formulas below,
	% A appears as reciprocal

	% Solution through (0,4),
	% A = -4/3
	\draw[thick, blue, domain=1.172:4.2, smooth] plot
	  ({- 3 * \x / (1-\x) / 4 - 1},\x);

  	% Solution through (0,0.25),
	% A = 1/3
	\draw[thick, red, domain=0.2:0.63, smooth] plot
	({3 * \x / (1-\x) - 1},\x);

	% Solution through (0,-0.1),
	% A = -1/11
	\draw[thick, purple, domain=-0.075:-0.87, smooth] plot
	  ({- 11 * \x / (1-\x) - 1},\x);

	% Solution through (0,-0.5),
	% A = -1/3
	\draw[thick, green!50!black, domain=-0.35:-4.2, smooth] plot
	  ({- 3 * \x / (1-\x) - 1},\x);



\end{tikzpicture}

\newcommand*{\xmin}{-0.5}
\newcommand*{\xmax}{6.5}
\newcommand*{\ymin}{-0.5}
\newcommand*{\ymax}{6.5}

%
% to maintain all segments at same length L
% use
%   delta-x = (L/2) / sqrt(1 + m^2)
%   delta-y = m * (L/2) / sqrt(1 + m^2)
% (the L/2 is because we draw two segments, each from centre out to either right or left)
%

% the rate function
\newcommand*{\dfslope}{- \y / 2}
%
\newcommand*{\halfsegmentlen}{0.15}  % (L = 0.3)
\newcommand*{\dfdeltax}{\halfsegmentlen/sqrt(1+(\dfslope)^2)}
\newcommand*{\dfdeltay}{\dfslope * \halfsegmentlen/sqrt(1+(\dfslope)^2)}

\begin{tikzpicture}

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[right] {$q(t)$};

	\foreach \x in {0,0.5,1,...,6} {
		\foreach \y in {0,0.5,1,...,6} {
			\draw (\x,\y) -- ++($({\dfdeltax},{\dfdeltay})$);
			\draw (\x,\y) -- ++($({-\dfdeltax},{-\dfdeltay})$);
		}
	}

\end{tikzpicture}

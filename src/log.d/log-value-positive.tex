\newcommand*{\xmin}{-0.5}
\newcommand*{\xmax}{5.5}
\newcommand*{\ymin}{-2.5}
\newcommand*{\ymax}{2.5}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt}
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0);
	\draw[->,thick] (0,\ymin) -- (0,\ymax);

	% graph
	\draw[thick, green!33!black, domain=0.1:5, smooth] plot (\x,{ln \x});
	\node[green!33!black,closedpt] at (4,1.386) (value) {};
	\draw[dashed] (4,0) -- (value) -- (0,1.386);
	\node[below] at (4,0) {$t$};
	\node[left] at (0,1.386) {$\ln t$};

\end{tikzpicture}

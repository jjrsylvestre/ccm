\newcommand*{\ymin}{-4}
\newcommand*{\ymax}{6.5}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt},
	xscale=2
]

	\foreach \x in {-1,1} {
		\draw[dashed,black!66!white] (\x,\ymin) to (\x,\ymax) node[above] {$t = \x$};
	}
	\draw[dashed,black!66!white] (\xmin,2) to node[below,at end,xshift=-0.5cm] {$y = 2$} (\xmax,2);

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[above right] {$q(t)$};

	\draw[blue,thick,domain=\xmin:-1.201,smooth] plot (\x,{2 * \x * \x / (\x * \x - 1)});
	\draw[blue,thick,domain=-0.817:0.817,smooth] plot (\x,{2 * \x * \x / (\x * \x - 1)});
	\draw[blue,thick,domain=1.201:\xmax,smooth] plot (\x,{2 * \x * \x / (\x * \x - 1)});

	\node[blue,closedpt] at (0,0) {};
	\node[above right,align=center] at (0,0) {local \\ max};

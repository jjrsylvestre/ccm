\newcommand*{\xmin}{-8}
\newcommand*{\xmax}{8}
\newcommand*{\ymin}{-3}
\newcommand*{\ymax}{2.25}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt},
	xscale=0.75,
	yscale=1.5
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax) node[right] {$q(t) = \ln \abs{t}$};

	% graph
	\draw[thick, blue, domain=0.05:7.75, smooth] plot (\x,{ln \x});
	\draw[thick, blue, domain=0.05:7.75, smooth] plot (- \x,{ln \x});

\end{tikzpicture}

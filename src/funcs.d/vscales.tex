\newcommand*{\xmin}{-0.5}
\newcommand*{\xmax}{6}
\newcommand*{\ymin}{-0.5}
\newcommand*{\ymax}{5}

\begin{tikzpicture}
[
	closedpt/.style={circle,draw,fill,inner sep=0pt,minimum size=4pt},
	openpt/.style={circle,draw,inner sep=0pt,minimum size=4pt},
]

	% axes
	\draw[->,thick] (\xmin,0) -- (\xmax,0) node[below] {$t$};
	\draw[->,thick] (0,\ymin) -- (0,\ymax);

	% graph
	\draw[thick, domain=0:2.44, smooth] plot ({\x * \x},\x) node[above] {$f(t)$};
	\draw[thick, blue, domain=0:2.44, smooth] plot ({\x * \x}, 2 * \x) node[below] {$g(t)$};
	\draw[thick, red, domain=0:2.44, smooth] plot ({\x * \x}, \x / 2) node[below] {$h(t)$};

	\node[closedpt] at (0,0) (orig0) {};
	\node[closedpt] at (1,1) (orig1) {};
	\node[closedpt,blue] at (1,2) (up1) {};
	\node[closedpt,red] at (1,0.5) (down1) {};

	\node[below left] at (orig0) {$(0,0)$};

	\node[above] at (orig1) {$(1,1)$};
	\node[above left,blue] at (up1) {$(1,2)$};
	\node[below right,red] at (down1) {$(1,1/2)$};

	\draw[->,blue] (3.5,2) -- (3.5,3.5);
	\draw[->,red] (3.5,1.75) -- (3.5,1.1);

\end{tikzpicture}
